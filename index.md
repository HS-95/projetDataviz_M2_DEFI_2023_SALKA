#Océans
SELECT DISTINCT ?ocean ?oceanLabel ?image
WHERE {
  ?ocean wdt:P31 wd:Q9430. # Un océan
  OPTIONAL { ?ocean wdt:P18 ?image.} #une image en lien avec l'océan

}
