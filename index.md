``` sparql
#
SELECT DISTINCT ?ocean ?oceanLabel
WHERE {
  ?ocean wdt:P31 wd:Q9430. #océans
  SERVICE wikibase:label {bd:serviceParam wikibase:language "fr" }
}
```
